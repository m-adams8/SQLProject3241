CREATE VIEW PublisherSales AS
SELECT P.Publisher_ID, p.Name AS PublisherName, COUNT(pur.ISBN) AS TotalOrders, SUM(b.Price * pur.Quantity) AS TotalRevenue
From Book b JOIN Are_Purchased pur ON b.ISBN = pur.ISBN JOIN Publisher p ON p.Publisher_ID = b.Publisher_ID
GROUP BY p.Publisher_ID, p.Name;

CREATE VIEW EmployeeOrderPerformance AS
SELECT e.Employee_ID, e.F_Name, e.L_Name, COUNT(o.Order_ID) AS OrdersHandled, AVG(o.Total) AS AvgOrderValue, SUM(o.Total) AS TotalSales
FROM Employee e JOIN Is_Fulfilled i ON e.Employee_ID = i.Employee_ID JOIN Orders o ON i.Order_ID = o.Order_ID
GROUP BY e.Employee_ID, e.F_Name, e.L_Name;

This view provides a performance summary for each employee by aggregating the orders they processed, including the count of orders, the average order value, and the total sales amount.
